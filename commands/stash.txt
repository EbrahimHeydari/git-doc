# دستورات انباشت (stash)

git stash                                           // تغییرات را به صورت موقت در جای دیگری انباشت می کند تا بعدا بازیابی بشن

git stash save 'message'                            // انباشت با نام دلخواه برای شناسایی راحت تر

git stash save -u                                   // انباشت فایل های ردیابی نشده (untracked)

git stash -a                                        // انباشت فایل نادیده گرفته شده (ignored)

git stash list                                      // لیست انباشت های موجود را نشان می دهد

git stash apply                                     // بازیابی آخرین انباشت

git stash apply [stash id]                          // بازیابی انباشت مشخص

git stash pop                                       // بازیابی و حذف آخرین انباشت

git stash pop [stash id]                            // بازیابی و حذف انباشت مشخص

git stash drop                                      // حذف آخرین انباشت

git stash drop [stash id]                           // حذف انباشت مشخص

git stash clear                                     // حذف تمامی انباشت ها

git stash show                                      // نمایش تغییرات آخرین انباشت

git stash show [stash id]                           // نمایش تغییرات انباشت مشخص

git stash show -p                                   // نمایش تغییرات تمامی انباشت ها

git stash branch [branch name]
میاد با آخرین انباشت ای که ما داریم یه برنچ میسازه و بعد انباشت رو حذف می‌کنه و به طور خودکار سوئیچ می‌کنه به برنچ جدیدی که ساخته شده
