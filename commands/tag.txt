# دستورات تگ (شماره نسخه)

git tag                                             // نمایش تگ ها (نسخه ها)

git tag -a v1.0 -m 'message'                        // ساخت تگ (نسخه)

git tag -d [tag name]                               // حذف تگ ها

git show [tag name]                                 // نمایش تغییرات در تگ مشخص
