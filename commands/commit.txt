# دستورات کامیت

git commit -am "message"                            // کامیت بدون استیج

git commit -a --allow-empty-message                 // کامیت بدون مسیج

git commit -m 'message'                             // ذخیره ی نهایی به همراه کامنت

git commit --amend -m "message"                     // ویرایش پیام آخرین کامیت

git commit --amend --no-edit                        // افزودن فایل به آخرین کامیت بدون نیاز به ثبت کامیت جدید

# بعد از استفاده از دستور بالا، قبل از ذخیره در ریموت، باید یکبار از دستور pull استفاده کنید

git revert [commit hash]                            // تمام تغییرات یک کامیت برمی گرداند و یک کامیت جدید ایجاد می کند
